WITH patient_abnormal_tests AS (
    SELECT 
        p.patient_id,
        p.patient_name,
        COUNT(DISTINCT lr.test_name) AS abnormal_test_count
    FROM patients p
    JOIN appointments a ON p.patient_id = a.patient_id
    JOIN lab_results lr ON a.appointment_id = lr.appointment_id
    WHERE lr.result_flag IN ('high', 'low', 'critical')
    GROUP BY p.patient_id, p.patient_name
    HAVING COUNT(DISTINCT lr.test_name) > 1
),
patient_balances AS (
    SELECT 
        p.patient_id,
        SUM(i.total_amount) - SUM(i.insurance_paid) AS outstanding_balance
    FROM patients p
    JOIN appointments a ON p.patient_id = a.patient_id
    JOIN invoices i ON a.appointment_id = i.appointment_id
    GROUP BY p.patient_id
),
avg_balance AS (
    SELECT AVG(outstanding_balance) AS avg_outstanding
    FROM patient_balances
)
SELECT 
    pat.patient_id,
    pat.patient_name
FROM patient_abnormal_tests pat
JOIN patient_balances pb ON pat.patient_id = pb.patient_id
