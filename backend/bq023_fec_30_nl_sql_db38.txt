Using indiv20, zipcode_to_census_tracts, census_tracts_new_york, and censustract_2018_5yr, list Kings County census tracts with the average 2020 individual donation amount and 2018 median income.

SQL:
SELECT c.tract_ce, c.tract_name
FROM census_tracts_new_york c
JOIN indiv20 i ON c.tract_ce = i.zip_code
WHERE i.transaction_dt LIKE '2020%'
GROUP BY c.tract_ce, c.tract_name

===== QUERY 17 =====
NL:
For each month in 2020, calculate New York individual contribution totals and the month-over-month percent change in total contribution amount.

SQL:
WITH monthly_totals AS (
  SELECT 
    strftime('%Y-%m', i.transaction_dt) AS month,
    SUM(i.transaction_amt) AS total_amt
  FROM individuals_ingest_2020 i
  JOIN candidate_committee_2020 c ON i.cmte_id = c.cmte_id
  JOIN census_tracts_new_york ct ON i.state = ct.state_fips_code
  WHERE i.transaction_dt LIKE '2020%'
    AND ct.state_fips_code = '36'
  GROUP BY strftime('%Y-%m', i.transaction_dt)
)
SELECT 
  month,
  total_amt,
  CASE 
    WHEN LAG(total_amt) OVER (ORDER BY month) IS NULL THEN NULL
    ELSE ((total_amt - LAG(total_amt) OVER (ORDER BY month)) * 100.0 / LAG(total_amt) OVER (ORDER BY month))
  END AS percent_change
FROM monthly_totals
ORDER BY month;

    FROM committee_contributions_2016
    GROUP BY cmte_id
),

    GROUP BY cmte_id

JOIN committee_2020 ON cte_2020.cmte_id = committee_2020.cmte_id;

===== QUERY 27 =====
NL:
Find pairs of committees that received donations from the same New York donor on the same transaction date, showing committee ids, donor name, date, and combined amount.

SELECT 
  t1.cmte_id AS cmte_id_1,
  t2.cmte_id AS cmte_id_2,

  FROM other_committee_tx_2016 WHERE state = 'NY'

  SELECT cmte_id, name, transaction_dt, transaction_amt, state

  FROM committee_contributions_2016 WHERE state = 'NY'
  UNION ALL
  SELECT cmte_id, name, transaction_dt, transaction_amt, state
  FROM committee_contributions_2018 WHERE state = 'NY'
  UNION ALL
  SELECT cmte_id, name, transaction_dt, transaction_amt, state
  FROM committee_contributions_2020 WHERE state = 'NY'
  UNION ALL
  SELECT cmte_id, name, transaction_dt, transaction_amt, state
  FROM indiv20 WHERE state = 'NY'

JOIN (
  SELECT cmte_id, name, transaction_dt, transaction_amt, state
  FROM other_committee_tx_2016 WHERE state = 'NY'
  UNION ALL
  SELECT cmte_id, name, transaction_dt, transaction_amt, state
  FROM other_committee_tx_2018 WHERE state = 'NY'

===== QUERY 28 =====
NL:
Find 2020 candidates whose New York donation total is above the average New York donation total across all candidates running for the same office.

SQL:
SELECT c.cand_id, c.cand_name, c.cand_office
FROM candidate_2020 c
WHERE c.cand_election_yr = 2020
  AND c.cand_office_st = 'NY'
  AND c.cand_pty_affiliation IS NOT NULL
  AND (
    SELECT SUM(CAST(c2.cand_pcc AS INTEGER))
    FROM candidate_2020 c2
    WHERE c2.cand_election_yr = 2020
      AND c2.cand_office_st = 'NY'
      AND c2.cand_id = c.cand_id
  ) > (
    SELECT AVG(candidate_total)
    FROM (
      SELECT SUM(CAST(c3.cand_pcc AS INTEGER)) AS candidate_total
      FROM candidate_2020 c3
      WHERE c3.cand_election_yr = 2020
        AND c3.cand_office_st = 'NY'
        AND c3.cand_office = c.cand_office
      GROUP BY c3.cand_id
    )
  );
