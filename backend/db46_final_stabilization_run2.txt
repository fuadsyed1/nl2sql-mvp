    SELECT p.patient_id, p.insurance_provider, SUM(i.total_amount) AS total_amount
    FROM patients p

),

    FROM patients p

          WHERE a2.patient_id = p.patient_id
      )
)
SELECT p.patient_id, p.patient_name
FROM patients p
JOIN patient_lifetime_totals plt ON p.patient_id = plt.patient_id
JOIN avg_by_provider abp ON plt.insurance_provider = abp.insurance_provider
JOIN most_recent_completed mrc ON p.patient_id = mrc.patient_id
