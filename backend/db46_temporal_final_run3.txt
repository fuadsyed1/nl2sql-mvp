WITH patient_specialty_counts AS (
    SELECT p.patient_id, COUNT(DISTINCT d.specialty) AS specialty_count
    FROM patients p
    JOIN appointments a ON p.patient_id = a.patient_id
    JOIN doctors d ON a.doctor_id = d.doctor_id
    GROUP BY p.patient_id
),
patient_invoices AS (
    SELECT p.patient_id, p.insurance_provider, SUM(i.total_amount) AS total_amount
    FROM patients p
    JOIN appointments a ON p.patient_id = a.patient_id
    JOIN invoices i ON a.appointment_id = i.appointment_id
    GROUP BY p.patient_id, p.insurance_provider
),
avg_by_provider AS (
    SELECT insurance_provider, AVG(total_amount) AS avg_total
    FROM patient_invoices
    GROUP BY insurance_provider
)
SELECT p.patient_id, p.patient_name
FROM patients p
JOIN patient_specialty_counts psc ON p.patient_id = psc.patient_id
JOIN patient_invoices pi ON p.patient_id = pi.patient_id
JOIN avg_by_provider abp ON pi.insurance_provider = abp.insurance_provider
WHERE psc.specialty_count > 1
  AND pi.total_amount > abp.avg_total
