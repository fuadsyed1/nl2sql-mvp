WITH patient_spending AS (
    SELECT a.patient_id, SUM(i.total_amount) AS total_spent
    FROM appointments a
    JOIN invoices i ON a.appointment_id = i.appointment_id
    GROUP BY a.patient_id
),
avg_patient_spending AS (
    SELECT AVG(total_spent) AS avg_spent
    FROM patient_spending
),
avg_test_value AS (
    SELECT test_name, AVG(test_value) AS avg_val
    FROM lab_results
    GROUP BY test_name
)
SELECT lr.lab_id, lr.test_name, lr.test_value, lr.result_flag, lr.result_date
FROM lab_results lr
JOIN appointments a ON lr.appointment_id = a.appointment_id
JOIN invoices i ON a.appointment_id = i.appointment_id
JOIN patients p ON a.patient_id = p.patient_id
JOIN patient_spending ps ON p.patient_id = ps.patient_id
JOIN avg_test_value atv ON lr.test_name = atv.test_name
CROSS JOIN avg_patient_spending aps
WHERE lr.test_value > atv.avg_val
  AND ps.total_spent > aps.avg_spent
