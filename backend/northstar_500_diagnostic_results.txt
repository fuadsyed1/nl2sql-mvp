STATUS: PASS
QUERY:
Calculate the percentage of sales orders that were delivered for each priority level.
SQL:
SELECT priority, (delivered_count * 100.0 / total_count) AS delivered_percentage
FROM (
    SELECT 
        priority,
        SUM(CASE WHEN fulfillment_status = 'delivered' THEN 1 ELSE 0 END) AS delivered_count,
        COUNT(order_id) AS total_count
    FROM sales_orders
    GROUP BY priority
)
WHERE delivered_count * 100.0 / total_count > (
    SELECT AVG(delivered_percentage)
    FROM (
        SELECT 
            priority,
            SUM(CASE WHEN fulfillment_status = 'delivered' THEN 1 ELSE 0 END) * 100.0 / COUNT(order_id) AS delivered_percentage
        FROM sales_orders
        GROUP BY priority
    )
)

WITH monthly_revenue AS (

QUERY:
For each product category, calculate its share of total inventory value and its share of total sales revenue.
SQL:
WITH inv_agg AS (
    SELECT p.category, SUM(i.inventory_value) AS total_inv_value
    FROM products p
    JOIN inventory i ON p.product_id = i.product_id
    GROUP BY p.category
),
sales_agg AS (
    SELECT p.category, SUM(soi.line_total) AS total_sales_revenue
    FROM products p
    JOIN sales_order_items soi ON p.product_id = soi.product_id
    GROUP BY p.category
),
totals AS (
    SELECT 
        SUM(total_inv_value) AS global_inv,
        SUM(total_sales_revenue) AS global_sales
    FROM (
        SELECT COALESCE(inv.total_inv_value, 0) AS total_inv_value,
               COALESCE(sales.total_sales_revenue, 0) AS total_sales_revenue
        FROM inv_agg inv
        FULL OUTER JOIN sales_agg sales ON inv.category = sales.category
    )
)
SELECT 
    COALESCE(inv.category, sales.category) AS category,
    COALESCE(inv.total_inv_value, 0) * 1.0 / t.global_inv AS inventory_share,
    COALESCE(sales.total_sales_revenue, 0) * 1.0 / t.global_sales AS sales_share
FROM inv_agg inv
FULL OUTER JOIN sales_agg sales ON inv.category = sales.category
CROSS JOIN totals t
WHERE COALESCE(inv.total_inv_value, 0) > 0 OR COALESCE(sales.total_sales_revenue, 0) > 0
