    SELECT 
        f.food_type,
        p.owner_id,
        p.unit_price,
        ROW_NUMBER() OVER (PARTITION BY f.food_type ORDER BY p.unit_price ASC) as rn_min,

    FROM ranked_purchases
    WHERE rn_min = 1
