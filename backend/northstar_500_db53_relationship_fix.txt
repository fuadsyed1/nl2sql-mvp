QUERY:
Calculate the refund rate for each payment method using refunded amount divided by payment amount.
SQL:
WITH method_totals AS (
  SELECT
    payment_method,
    SUM(refunded_amount) AS total_refunded,
    SUM(amount) AS total_amount
  FROM payments
  GROUP BY payment_method
),
global_avg AS (
  SELECT
    AVG(total_refunded) AS avg_refunded
  FROM method_totals
)
SELECT
  t.payment_method
FROM method_totals t
CROSS JOIN global_avg g
WHERE t.total_refunded > g.avg_refunded

WITH item_margin AS (
    SELECT
        p.category,
        SUM(soi.line_total - (soi.quantity_ordered * soi.unit_cost)) AS total_margin,
        SUM(soi.line_subtotal) AS total_subtotal
    FROM products p
    JOIN sales_order_items soi ON p.product_id = soi.product_id
    GROUP BY p.category
)

    category,
    (total_margin * 100.0 / total_subtotal) AS avg_gross_margin_pct
FROM item_margin
GROUP BY category
HAVING total_subtotal > 0

Calculate the revenue share of each product category as a percentage of total order-item revenue.
SQL:
WITH category_revenue AS (
    SELECT 
        p.category,
        SUM(soi.line_total) AS category_total
    FROM products p
    JOIN sales_order_items soi ON p.product_id = soi.product_id
    GROUP BY p.category
),
total_revenue AS (
    SELECT SUM(category_total) AS grand_total
    FROM category_revenue
)

    cr.category,
    (cr.category_total * 100.0 / tr.grand_total) AS revenue_share_pct
FROM category_revenue cr
CROSS JOIN total_revenue tr
WHERE tr.grand_total > 0
