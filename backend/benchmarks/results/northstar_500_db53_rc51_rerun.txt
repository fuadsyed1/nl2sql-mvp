CATEGORY: aggregation
DIFFICULTY: hard
STATUS: PASS
QUERY:
What is the average shipment weight for orders containing at least one Outdoors product?
SQL:
WITH order_weights AS (
  SELECT s.order_id, AVG(s.total_weight_kg) AS avg_weight
  FROM shipments s
  GROUP BY s.order_id
),
outdoors_orders AS (
  SELECT DISTINCT soi.order_id
  FROM sales_order_items soi
  JOIN products p ON soi.product_id = p.product_id
  WHERE p.category = 'Outdoors'
)
SELECT AVG(ow.avg_weight) AS average_weight
FROM order_weights ow
JOIN outdoors_orders oo ON ow.order_id = oo.order_id
GROUP BY oo.order_id
HAVING COUNT(DISTINCT oo.order_id) > 0

FROM products p
JOIN sales_order_items soi ON p.product_id = soi.product_id
JOIN sales_orders so ON soi.order_id = so.order_id
GROUP BY p.product_id, p.product_name
HAVING SUM(CASE WHEN strftime('%Y', so.order_date) = '2025' THEN soi.line_total ELSE 0 END) >

QUERY:
List carriers whose on-time delivery rate is above the overall carrier average and whose average shipping cost is below the overall average shipping cost.
SQL:
WITH carrier_stats AS (
    SELECT 
        carrier,
        SUM(CASE WHEN actual_delivery_date <= estimated_delivery_date THEN 1 ELSE 0 END) AS on_time_count,
        COUNT(*) AS total_shipments,
        SUM(shipping_cost) AS total_shipping_cost
    FROM shipments
    GROUP BY carrier
),
overall_averages AS (
    SELECT 
        AVG(on_time_count * 1.0 / total_shipments) AS avg_on_time_rate,
        AVG(total_shipping_cost * 1.0 / total_shipments) AS avg_shipping_cost
    FROM carrier_stats
)
SELECT 
    cs.carrier
FROM carrier_stats cs
CROSS JOIN overall_averages oa
WHERE (cs.on_time_count * 1.0 / cs.total_shipments) > oa.avg_on_time_rate
  AND (cs.total_shipping_cost * 1.0 / cs.total_shipments) < oa.avg_shipping_cost
